\subsection*{Characters}

\par\noindent
Let $V$ be a finite-dimensional complex representation of a group $G$.\n

\defn{
	The \textbf{character} of $V$ is the function $\map{\mathcal{X}_{V}:G}{\C}{g}{\tr(g|_{V})}$.\n
}

\lemma{
	\begin{itemize}[leftmargin=4\parindent, itemsep=1.5pt, topsep=0pt]
		\item $\mathcal{X}_{V\oplus{}W}=\mathcal{X}_{V}+\mathcal{X}_{W}$
		\item $\mathcal{X}_{\Hom(V,W)}=\conj{\mathcal{X}_{V}}\mathcal{X}_{W}$
		\item $\mathcal{X}_{V^{*}}=\conj{\mathcal{X}_{V}}$
	\end{itemize}\up\n
}

\lemma{
	$\mathcal{X}_{V}$ satisfies $\mathcal{X}_{V}(ghg\inv)=\mathcal{X}_{V}(h)$, $\forall{}g,h\in{}G$.\n
}

\defn{
	A \textbf{class function} on $G$ is a function $f:G\to\C$ \st{} $f(ghg\inv)=f(h)$, $\forall{}g,h\in{}G$.\n
}

\ex{
	$\mathcal{X}_{V}$ is a class function.\n
}

\defn{
	$\mathscr{C}(G)$ is the space of all class functions. It is a $\C$-vector space, and its dimension is the number of conjugacy classes in $G$.\n
}

\par\noindent
For $\varphi,\psi\in\mathscr{C}(G)$, define
\[
	\iprod{\varphi,\psi}=\frac{1}{\#G}\sum_{g\in{}G}\conj{\varphi(g)}\psi(g)
\]
This is a positive definite Hermitian inner product on $\mathscr{C}(G)$.\n

\defn{
	$\mathcal{X}_{\triv}=\ones\in\mathscr{C}(G)$ is the function $g\mapsto{}1$, $\forall{}g\in{}G$.\n
}

\thm{
	If $V$ is a representation of $G$, then $\dim{}V^{G}=\iprod{\ones,\mathcal{X}_{V}}$.\n
	Idea of a proof: Define $\pi:V\to{}V$ where $v\mapsto\frac{1}{\#G}\sum_{g\in{}G}gv$.\n
	Clearly, $\tr(\pi)=\iprod{\ones,\mathcal{X}_{V}}$. On the other hand, $\pi^{2}=\pi$, and $\im(\pi)=V^{G}$, so $\tr(\pi)=\dim{}V^{G}$.\n
}

\cor{
	For $V,W$ representations of $G$, $\dim{}\Hom_{G}(V,W)=\iprod{\mathcal{X}_{V},\mathcal{X}_{W}}$.\n
	Proof: Recall $\Hom_{G}(V,W)=\Hom(V,W)^{G}$. By the theorem,
	\[
		\dim\Hom_{G}(V,W)=\iprod{\ones,\mathcal{X}_{\Hom(V,W)}}=\iprod{\ones,\conj{\mathcal{X}_{V}}\mathcal{X}_{W}}=\iprod{\mathcal{X}_{V},\mathcal{X}_{W}}
	\]
	\proven
}

\newpage
\thm{
	(Schur's Lemma) For $V,W$ irreducible representations of $G$,
	\[
		\dim\Hom_{G}(V,W)=\left\{\begin{array}{ll}
			0 & V\not\cong{}W\\
			1 & V\cong{}W
		\end{array}\right.
	\]
	Moreover, if $V=W$, $\Hom_{G}(V,V)=\vspan(\Id_{V})$.\n
}

\cor{
	(Schur Orthogonality) If $V,W$ are irreducible representations, then
	\[
		\iprod{\mathcal{X}_{V},\mathcal{X}_{W}}=\left\{\begin{array}{ll}
			0 & V\not\cong{}W\Leftrightarrow\mathcal{X}_{V}\ne\mathcal{X}_{W}\\
			1 & V\cong{}W\Leftrightarrow\mathcal{X}_{V}=\mathcal{X}_{W}
		\end{array}\right.
	\]\up\n
}

\par\noindent
This implies the number of irreducible representations (up to isomorphism) is at most $\dim\mathscr{C}(G)$, which in turn is the number of conjugacy classes of $G$.\n

\par\noindent
Let $L_{1},\ldots,L_{r}$ be ``the'' irreducible representations of $G$ (that is, choose one from each isomorphism class). Any representation $V$ can be decomposed as
\[
	V\cong{}L_{1}^{\oplus{}m_{1}}\oplus{}L_{2}^{\oplus{}m_{2}}\oplus\cdots\oplus{}L_{r}^{\oplus{}m_{r}}
\]

\defn{
	$m_{i}$ is called the \textbf{multiplicity} of $L_{i}$ in $V$.\n
}

\par\noindent
$\mathcal{X}_{V}=m_{1}\mathcal{X}_{1}+\cdots+m_{r}\mathcal{X}_{r}$. By Schur Orthogonality, $m_{i}=\iprod{\mathcal{X}_{i},\mathcal{X}_{r}}$.\n

\cor{
	If $\mathcal{X}_{V}=\mathcal{X}_{W}$, then $V\cong{}W$.\n
}
