\par\noindent
Recall: $P=\R^{2}$ is the plane. $M$ is the group of rigid motions of $P$.\n
From last time, we have $M$ generated by $t_{a}$ translations, $\rho_{\theta}$ rotations about $0$, and $r$ reflections across the $x$-axis.\n

\defn{
	Let $S\subset{}P$ be a ``planar figure''. The \textbf{symmetry group} of $S$ is $\Gamma_{S}=\set{m\in{}M\mid{}mS=S}$. This is a subgroup of $M$.\n
}

\ex{
	\begin{enumerate}
		\item Let $S$ be the unit circle. $\Gamma_{S}=O(2)$ rotations and reflections through the origin.
		\item $S=\R^{2}\Rightarrow\Gamma_{S}+M$.
		\item $S$ is the unit circle with an arrow at $\frac{2\pi{}k}{n}$ for each $k$. Then $\Gamma_{S}=\set{\ptxt{rotations by }\frac{2\pi{}k}{n}}\cong\Z/n\Z$.
		\item $S$ a regular $n$-gon. We have rotations by $\frac{2\pi{}k}{n}$, for $0\le{}k\le{}n-1$, reflect through certain lines (lines that go through the origin and a vertex or midpoint of an edge).
	\end{enumerate}
}

\defn{
	For $S$ a regular $n$-gon, $\Gamma_{S}$ is denoted $D_{n}$ and called the \textbf{$n$th dihedral group}. It's generated by $a=\rho_{\frac{2\pi}{n}}$ and $b=r$.\n
}

\par\noindent
Relations: $a^{n}=1$, $b^{2}=1$, $bab\inv=a\inv$. Every element of $D_{n}$ can be written uniquely as $a^{k}$ or $a^{k}b$ for $0\le{}k<n$. Note that $\card{D_{n}}2n$, and $D_{3}\cong{}S_{3}$.\n

\par\noindent
Question: can we classify the symmetry groups of plane figures? A good first step is classifying the finite subgroups of $M$.\n

\thm{
	Suppose $G$ is a finite subgroup of $M$. Then $G$ has a fixed point, i.e., $\exists{}x\in{}P$ \st{} $gx=x,\forall{}g\in{}G$.\n
	Proof: Let $S$ be a finite subset of $P$. Define the center of gravity of $S$ by $COG(S)=\frac{1}{n}\sum_{i=1}^{n}x_{i}$.\n
}

\lemma{
	Given $m\in{}S$, $COg(mS)=m(COG(s))$ (note: $mS=\set{mx_{1},\ldots,mx_{n}}$).\n
	Proof: Let $m\in{}O(2)$. Then $mCOG(s)=m(\frac{1}{n}\sum_{i=1}^{n}x_{i})=\frac{1}{n}\sum_{i=1}^{n}mx_{i}=COG(mS)$. Let $m=t_{a}$. Then $COG(mS)=\frac{1}{n}\sum_{i=1}^{n}(x_{i}+a)=\frac{1}{n}\sum_{i=1}^{n}x_{i}+a=COG(S)+a=mCOG(s)$.\proven
}

\par\noindent\settowidth{\hangindent}{\textbf{Thm: }}\phantom{\textbf{Thm: }}Continued: pick some $x_{0}\in{}P$ at random. Let $S=\set{gx_{0}\mid{}g\in{}G}$. Observe that for $h\in{}G$, $hS=S$. $hS=\set{hgx_{0}\mid{}g\in{}G}=\set{gx_{0}\mid{}g\in{}G}$. So $hCOG(S)=COS(hS)=COG(S)$. So $x=COS(S)$ is a fixed point of $G$.\proven

\par\noindent
Suppose $G\subset{}M$ is a subgroup that fixes $x\in{}P$. Then $t_{x}\inv{}Gt_{x}$ fixes $0\in{}P$, becuase, for $g\in{}G$,
\[
	(tx\inv{}gt_{x})0=t_{x}\inv(g(t_{x}(0)))=t_{x}\inv(g(x))=t_{x}\inv(x)=0
\]

\cor{
	Let $G\subset{}M$ be a finite subgroup. Then $\exists{}x\in{}P$ \st{} $t_{x}Gt_{x}\inv\subset{}O(2)$.\n
}

\par\noindent
Problem: classify the finite subgroups of $O(2)$.\n

\defn{
	$SO(2)\subseteq{}O(2)$ is the group of rotations.\n
}

\par\noindent
Note: every element of $O(2)$ can be written as $\rho_{\theta}$ or $\rho_{\theta}r$. This means $1$ and $r$ are the coset representatives for $O(2)$ with respect to $SO(2)$, so $[O(2):SO(2)]=2$.\n
First step: classify finite subgroups of $SO(2)$.\n

\prop{
	Let $G$ be a finite subgroup of $SO(2)$. Then $G$ is cyclic, and generated by $\rho_{\theta}$, with $\theta=\frac{2\pi}{n}$ for some $n\in\N$.\n
	Proof: let $0<\theta$ be minimal such that $\rho_{\theta}\in{}G$. Then because $G$ is finite, $\rho_{\theta}$ has finite order, so $\theta$ is a rational multiple of $2\pi$.\n
	We can write $\theta=\frac{2\pi{}p}{q}$, with $p,q$ real, coprime, and positive. $\rho_{\theta}$ then generates $\rho_{\frac{2\pi}{q}}$, because we can write $ab+pq=1$ for some $a,b\in\Z$, so $\rho_{\frac{2\pi}{q}}=\rho_{\frac{2\pi}{q}}^{ap+bq}=(\rho_{\frac{2\pi{}p}{q}})^{a}$.\n
	By the minimality of $\rho$, $\theta=\frac{2\pi}{q}$. So $G$ is generated by $\rho_{\theta}$; suppose $\rho_{\psi}\in{}G$, with $\psi\ge{}0$. Pick $n$ \st{} $\rho_{\psi}\rho_{\theta}^{-n}$ is a rotation by an angle in $[0,\frac{2\pi}{q})$. By minimality of $\theta$, $\rho_{\psi}\rho_{\theta}^{-n}\Rightarrow\rho_{\psi}=\rho_{\theta}^{n}$.\proven
}

\prop{
	Let $G$ be a finite subgroup of $O(2)$. Then either $G$ is cyclic and generated by $\rho_{\frac{2\pi}{n}}$, or $G$ is dihedral.\n
	Proof: Case 1: $G$ does not contain a reflection. Then $G\subset{}SO(2)$. So we're done by our classification of subgroups of $SO(2)$.\n
	Case 2: $G$ contains a reflection $b$. Let $H=G\cap{}SO(2)$. We know $H$ is cyclic, and generated by $\rho_{\frac{2\pi}{n}}$ for some $n$. If $g\in{}G$, then either $g\in{}H$ or $gb\in{}H$. Thus, $[G:H]=2$, so $G=D_{n}$.\n
	We conclude every finite subgroup of $M$ is either cyclic or dihedral.\proven
}
